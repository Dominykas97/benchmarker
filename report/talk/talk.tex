\begin{frame}{Execution}

          \begin{umlfragment}[type=loop, label={$\forall$ messages}, inner xsep=7]

          \end{umlfragment}
        \end{umlcall}
        \begin{umlcall}[op={runtime}, type=return]{Benchmarker}{ControlServer}
        \end{umlcall}

          \begin{umlcall}[op={metric}]{ControlServer}{PersistentVolume}
          \end{umlcall}
        \end{umlfragment}
      \end{umlseqdiag}
